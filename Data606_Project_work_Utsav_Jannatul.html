The stratified 10-fold cross-validation to find the cross-validation errors of the classification tree:

library(caret)
set.seed(10)
folds<-createFolds(housing_price_data$priceGroup, k=10)
library(MASS)

misclassification.tree<-function(dt){
  train<-housing_price_data[dt,]
  test<-housing_price_data[-dt,]
  fit<-tree(priceGroup~factor(Neighborhood) +factor(LotConfig)+factor(RoofMatl) +PoolArea+factor(MSSubClass)+factor(SaleCondition) +factor(BsmtQual) +factor(BsmtExposure) +factor(LandContour) +MoSold +factor(KitchenQual) +BsmtFinSF1 +GarageCars +LotArea +OverallQual +OverallCond +GrLivArea +TotalBath, train)
  pred<-predict(fit,test, type = "class")
  return(1-mean(pred==test$priceGroup))
}
mis_rate.tree=lapply(folds,misclassification.tree)
